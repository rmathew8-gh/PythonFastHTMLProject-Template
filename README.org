* Setup

#+BEGIN_SRC bash
# Install dependencies
make install

# Or manually with uv
uv sync
#+END_SRC

* Development

#+BEGIN_SRC bash
# Start development server
make dev

# Or manually
uv run python -m fasthtml dev
#+END_SRC

* Production

#+BEGIN_SRC bash
# Run production server
make run

# Or manually
uv run python -m fasthtml run
#+END_SRC

* Other Commands

#+BEGIN_SRC bash
# Lint code
make lint

# Run tests
make test

# Run integration tests
make test-integration

# Run all tests with coverage
make test-coverage

# Clean cache files
make clean
#+END_SRC

* Project Structure

- =app.py= - Main FastHTML application
- =pyproject.toml= - Project configuration and dependencies
- =Makefile= - Development tasks and test runner
- =tests/= - Comprehensive test suite including integration tests
