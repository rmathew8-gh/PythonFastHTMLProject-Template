* Pattern
#+BEGIN_SRC python
from fastapi import FastAPI, HTMLResponse
from fasthtml.common import Html, Head, Body, H1, P

app = FastAPI()

@app.get("/", response_class=HTMLResponse)
def hello():
    return str(Html(
        Head(Title("Title")),
        Body(H1("Hello"), P("Content"))
    ))
#+END_SRC
